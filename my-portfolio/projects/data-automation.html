Key Macro:

Sub GenerateReports()
    ' Connect to SQL database
    Dim conn As ADODB.Connection
    Set conn = New ADODB.Connection
    conn.Open "Provider=SQLOLEDB;Data Source=SchoolDB;"
    
    ' Pull attendance data
    Dim rs As ADODB.Recordset
    Set rs = conn.Execute("SELECT * FROM Attendance WHERE Month=" & MonthBox.Value)
    
    ' Generate PDF
    Sheets("Report").ExportAsFixedFormat Type:=xlTypePDF, _
        Filename:="Reports\Attendance_" & Format(Date, "yyyymm") & ".pdf"
    
    MsgBox "Report generated successfully!", vbInformation
End Sub
